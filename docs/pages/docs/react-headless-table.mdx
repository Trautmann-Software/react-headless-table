  <Tab>
    ```tsx filename="table.tsx"
    import { ReactHeadlessTable } from '@trautmann/react-headless-table';

    type MyDataStructure = {
      username: string;
      age: number;
    };

    const rows: Array<MyDataStructure> = [
      { username: 'user-foo', age: 17 },
      { username: 'user-bar', age: 42 }
    ];

    const columns: Array<Column<MyDataStructure>> = [
      { field: 'username', type: 'string' },
      { field: 'age', type: 'number' }
    ];

    function Table() {
      return (
        <ReactHeadlessTable<MyDataStructure>
          columns={columns}
          rows={rows}
        >
          <table>
            <TableHead />
            <TableBody />
          </table>
        </ReactHeadlessTable>
      );
    }
    ```

    ```tsx filename="table-head.tsx"
    import { useColumns } from '@trautmann/react-headless-table';

    function TableHead() {
      const { columns } = useColumns<MyDataStructure>();
      return (
        <thead>
          <tr>
            {columns.map((column) => (
              <td key={column.id}>{column.field}</td>
            ))}
          </tr>
        </thead>
      );
    }
    ```

    ```tsx filename="table-body.tsx"
    import { useColumns, useRows } from '@trautmann/react-headless-table';

    function TableBody() {
      const { columns } = useColumns<MyDataStructure>();
      const { rows } = useRows<MyDataStructure>();
      return (
        <tbody>
          {rows.map((row) => (
            <tr key={row.id}>
              {columns.map((column) => (
                <td key={column.id}>{row.data?.[column.field] ?? ''}</td>
              ))}
            </tr>
          ))}
        </tbody>
      );
    }
    ```
